function setupDriveFolder() {

function getRootFolder() {
  const folders = DriveApp.getFoldersByName(DRIVE_FOLDER_NAME);
  if (folders.hasNext()) return folders.next();
  return DriveApp.createFolder(DRIVE_FOLDER_NAME);
}

function getWilayah() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('MASTER_WILAYAH');
    if (sheet.getLastRow() <= 1) return [];
    
    const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 2).getValues();
    return data.map(row => ({ 
      id: String(row[0] || ''), 
      nama: String(row[1] || '') 
    })).filter(w => w.id);
  } catch (error) {
    Logger.log("Error getWilayah: " + error);
    return [];
  }
}

function getAnggotaByWilayah(wilayahTarget) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('MASTER_ANGGOTA');
    if (!sheet || sheet.getLastRow() <= 1) return [];
    
    const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 3).getValues();
    const anggota = [];
    for (let i = 0; i < data.length; i++) {
      if (String(data[i][2]) === String(wilayahTarget)) {
        anggota.push({ id: String(data[i][0] || ''), nama: String(data[i][1] || '') });
      }
    }
    return anggota.sort((a, b) => a.nama.localeCompare(b.nama));
  } catch (error) {
    Logger.log("Error getAnggotaByWilayah: " + error);
    return [];
  }
}

function getKegiatan() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('KEGIATAN');
    if (sheet.getLastRow() <= 1) return [];
    
    const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 7).getValues();
return data.map(row => ({
  id: String(row[0] || ''),
  nama: String(row[1] || ''),
  tanggal: (row[2] instanceof Date) ? Utilities.formatDate(row[2], 'Asia/Jakarta', 'yyyy-MM-dd HH:mm') : String(row[2] || ''),
  lokasi: String(row[3] || ''),
  status: String(row[4] || ''),
  latitude: row[5] !== '' ? parseFloat(row[5]) : null,   // Kolom F
  longitude: row[6] !== '' ? parseFloat(row[6]) : null   // Kolom G
})).filter(k => k.id).reverse();
  } catch (error) {
    Logger.log("Error getKegiatan: " + error);
    return [];
  }
}

function getKegiatanAktif() {
  const kegiatan = getKegiatan();
  return kegiatan.find(k => k.status === 'Aktif') || null;
}

function submitAbsensi(formData) {
  try {
    const nama = String(formData.nama || '').trim();
    const wilayah = String(formData.wilayah || '').trim();
    
    if (!nama) return { success: false, message: 'Nama wajib diisi.' };
    if (!wilayah) return { success: false, message: 'Wilayah wajib dipilih.' };

    const kegiatan = getKegiatanAktif();
    const namaKegiatan = kegiatan ? String(kegiatan.nama).trim() : 'Absensi Harian (Tanpa Kegiatan)';

    // TAMBAHAN: Mencegah absensi sebelum hari H dan sebelum JAM acara
    if (kegiatan && kegiatan.tanggal) {
      // Format waktu hari ini disamakan dengan format jam (Tahun-Bulan-Hari Jam:Menit)
      const waktuHariIni = Utilities.formatDate(new Date(), 'Asia/Jakarta', 'yyyy-MM-dd HH:mm');
      
      if (waktuHariIni < kegiatan.tanggal) {
        return { success: false, message: 'Gagal: Absensi belum dibuka. Acara baru dimulai pada ' + kegiatan.tanggal };
      } 
      
      // Untuk batas penutupan, kita potong dan cukup cek tanggalnya saja 
      // agar tidak tertutup otomatis sesaat setelah jam 10:00 lewat.
      const tglHariIni = waktuHariIni.substring(0, 10);
      const tglKegiatan = kegiatan.tanggal.substring(0, 10);
      if (tglHariIni > tglKegiatan) {
        return { success: false, message: 'Gagal: Absensi sudah ditutup. Acara ini berlangsung pada ' + tglKegiatan };
      }
    }

    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
        const sheet = ss.getSheetByName('ABSENSI');
        // Pindahkan pembuatan waktu (Tanggal) ke atas agar bisa dicek sebagai syarat pencegah duplikat
        const now = new Date();
        const timestamp = Utilities.formatDate(now, 'Asia/Jakarta', 'yyyy-MM-dd HH:mm:ss');
        const tanggal = Utilities.formatDate(now, 'Asia/Jakarta', 'dd/MM/yyyy');
        const jam = Utilities.formatDate(now, 'Asia/Jakarta', 'HH:mm:ss');
        // Pengecekan duplikat diperketat: Cek Tanggal + Nama + Wilayah + Kegiatan
        if (sheet.getLastRow() > 1) {
          const existing = sheet.getRange(2, 1, sheet.getLastRow() - 1, 6).getValues();
          const duplikat = existing.find(row => {
            // Amankan pembacaan tanggal (mencegah error jika Google Sheets otomatis merubahnya jadi kalender)
            let tglSheet = row[1];
            if (tglSheet instanceof Date) {
              tglSheet = Utilities.formatDate(tglSheet, 'Asia/Jakarta', 'dd/MM/yyyy');
            } else {
              tglSheet = String(tglSheet).trim();
            }
            
            return tglSheet === tanggal &&
                   String(row[3]).toLowerCase().trim() === nama.toLowerCase().trim() &&
                   String(row[4]).trim() === wilayah.trim() &&
                   String(row[5]).trim() === namaKegiatan.trim();
          });
          if (duplikat) {
            return { success: false, message: 'Gagal: Nama ' + nama + ' sudah tercatat hadir hari ini!' };
          }
        }

        sheet.appendRow([timestamp, tanggal, jam, nama, wilayah, namaKegiatan]);
        return {
      success: true,
      message: 'Absensi berhasil!',
      data: { nama: nama, wilayah: wilayah, kegiatan: namaKegiatan, jam: jam }
    };
  } catch (error) {
    return { success: false, message: 'Gagal memproses absensi: ' + error.message };
  }
}

// ============================================================

function performOCR(base64Data, mimeType) {
  try {
    const cleanBase64 = base64Data.replace(/^data:image\/[a-z]+;base64,/, "");
    const props = PropertiesService.getScriptProperties();
    const apiKey = props.getProperty('GEMINI_API_KEY');
    const rawModel = props.getProperty('YASSA_AI_MODEL') || 'models/gemini-2.5-flash'; 
    
    if (!apiKey) {
      return { success: false, message: 'Sistem Gagal: GEMINI_API_KEY belum disetel di Script Properties.' };
    }

    const modelPath = rawModel.startsWith('models/') ? rawModel : 'models/' + rawModel;
    const url = `https://generativelanguage.googleapis.com/v1beta/${modelPath}:generateContent?key=${apiKey}`;
    
    const promptText = "Tolong baca teks dalam gambar catatan/notulen ini. Perbaiki ejaan yang berantakan, lalu buatkan rangkuman yang rapi dalam bentuk poin-poin terstruktur yang terdiri dari: \n1. Agenda \n2. Keputusan \n3. Tindak Lanjut. \nPastikan outputnya profesional, akurat, dan abaikan coretan/informasi yang tidak relevan.";

    const payload = {
      "contents": [{
        "parts": [
          { "text": promptText },
          {
            "inlineData": {
              "mimeType": mimeType,
              "data": cleanBase64
            }
          }
        ]
      }]
    };

    const options = {
      "method": "post",
      "contentType": "application/json",
      "payload": JSON.stringify(payload),
      "muteHttpExceptions": true
    };

    const response = UrlFetchApp.fetch(url, options);
    const json = JSON.parse(response.getContentText());

    if (json.error) {
      throw new Error(json.error.message);
    }

    const hasilRangkuman = json.candidates[0].content.parts[0].text;
    
    return { 
      success: true, 
      text: hasilRangkuman 
    };

  } catch (error) {
    Logger.log('Error performOCR: ' + error.toString());
    return { 
      success: false, 
      message: 'AI Gagal memproses gambar: ' + error.message 
    };
  }
}

function addKegiatan(data) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('KEGIATAN');
    
    const idKegiatan = 'KGT-' + new Date().getTime();

    // Ambil nilai latitude & longitude dari form, jika kosong simpan string kosong
    const lat = (data.latitude !== undefined && data.latitude !== '') ? parseFloat(data.latitude) : '';
    const lng = (data.longitude !== undefined && data.longitude !== '') ? parseFloat(data.longitude) : '';

    // Urutan kolom: ID | Nama | Tanggal | Lokasi | Status | Latitude | Longitude
    sheet.appendRow([idKegiatan, data.nama, data.tanggal, data.lokasi, 'Aktif', lat, lng]);
    
    return { success: true, message: 'Kegiatan berhasil ditambahkan!' };
  } catch (error) {
    return { success: false, message: 'Gagal menambah kegiatan: ' + error.message };
  }
}

function updateKegiatan(data) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('KEGIATAN');
    const rows = sheet.getDataRange().getValues();
    
    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === data.id) {
        sheet.getRange(i + 1, 2).setValue(data.nama);
        sheet.getRange(i + 1, 3).setValue(data.tanggal);
        sheet.getRange(i + 1, 4).setValue(data.lokasi);
        // Simpan koordinat ke kolom F (kolom ke-6) dan G (kolom ke-7)
        const lat = (data.latitude !== undefined && data.latitude !== '') ? parseFloat(data.latitude) : '';
        const lng = (data.longitude !== undefined && data.longitude !== '') ? parseFloat(data.longitude) : '';
        sheet.getRange(i + 1, 6).setValue(lat);
        sheet.getRange(i + 1, 7).setValue(lng);
        return { success: true, message: 'Kegiatan berhasil diperbarui!' };
      }
    }
    return { success: false, message: 'Kegiatan tidak ditemukan.' };
  } catch (error) {
    return { success: false, message: 'Gagal memperbarui kegiatan: ' + error.message };
  }
}

function updateKegiatanStatus(id, status) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('KEGIATAN');
    const rows = sheet.getDataRange().getValues();
    
    if (status === 'Aktif') {
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][4] === 'Aktif') {
          sheet.getRange(i + 1, 5).setValue('Selesai');
        }
      }
    }

    for (let i = 1; i < rows.length; i++) {
      if (rows[i][0] === id) {
        sheet.getRange(i + 1, 5).setValue(status);
        return { success: true, message: 'Status berhasil diubah!' };
      }
    }
    return { success: false, message: 'Kegiatan tidak ditemukan.' };
  } catch (error) {
    return { success: false, message: 'Gagal mengubah status: ' + error.message };
  }
}

function deleteKegiatan(id) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('KEGIATAN');
    const rows = sheet.getDataRange().getValues();
    
    for (let i = 1; i < rows.length; i++) {
      if (String(rows[i][0]) === String(id)) {
        // [SESUDAH] 1. Catat nama kegiatan sebelum barisnya dihapus (kolom ke-2)
        const namaKegiatan = String(rows[i][1]).trim();

        // 2. Hapus kegiatan dari sheet KEGIATAN
        sheet.deleteRow(i + 1); 
        
        // 3. Buka sheet ABSENSI dan hapus semua data yang punya nama kegiatan tersebut
        const absenSheet = ss.getSheetByName('ABSENSI');
        if (absenSheet && absenSheet.getLastRow() > 1) {
          const absenRows = absenSheet.getDataRange().getValues();
          // Kita harus menghapus dari baris paling bawah ke atas agar urutan baris tidak rusak
          for (let j = absenRows.length - 1; j >= 1; j--) {
            if (String(absenRows[j][5]).trim() === namaKegiatan) {
              absenSheet.deleteRow(j + 1);
            }
          }
        }

        return { success: true, message: 'Kegiatan & Absensi berhasil dihapus!' };
      }
    }
    return { success: false, message: 'Kegiatan tidak ditemukan.' };
  } catch (error) {
    return { success: false, message: 'Gagal menghapus kegiatan: ' + error.message };
  }
}

function generateAbsensiPDF(namaKegiatan) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const dataAbsensi = ss.getSheetByName('ABSENSI').getDataRange().getDisplayValues();
    
    let html = `<div style="font-family:sans-serif;">
      <h2 style="text-align:center;">Daftar Hadir Peserta</h2>
      <h3 style="text-align:center; color:#555; margin-top:-10px;">${namaKegiatan}</h3><hr><br>
      <table border="1" cellpadding="8" cellspacing="0" style="width:100%; border-collapse:collapse; font-size:13px;">
        <tr style="background-color:#f2f2f2;"><th>No</th><th>Waktu Kehadiran</th><th>Nama Lengkap</th><th>Wilayah</th></tr>`;
    
    let no = 1;
    let adaData = false;
    for (let i = 1; i < dataAbsensi.length; i++) {
      if (String(dataAbsensi[i][5]).trim() === String(namaKegiatan).trim() || String(namaKegiatan) === '') {
        html += `<tr><td align="center">${no++}</td><td>${dataAbsensi[i][0]}</td><td>${dataAbsensi[i][3]}</td><td>${dataAbsensi[i][4]}</td></tr>`;
        adaData = true;
      }
    }
    html += `</table></div>`;
    
    if (!adaData) return { success: false, message: 'Tidak ada data kehadiran untuk kegiatan ini.' };

    const blob = Utilities.newBlob(html, MimeType.HTML).getAs(MimeType.PDF).setName(`Absensi - ${namaKegiatan}.pdf`);
    const file = getRootFolder().createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    
    return { success: true, url: file.getUrl(), message: 'PDF Absensi berhasil dibuat' };
  } catch (error) {
    return { success: false, message: 'Gagal membuat PDF: ' + error.message };
  }
}

// PROXY NOMINATIM - Cari Lokasi via Server (Anti-iframe block)
// ============================================================
function cariLokasiProxy(query) {
  try {
    const url = 'https://nominatim.openstreetmap.org/search?q=' 
      + encodeURIComponent(query) 
      + '&format=json&limit=6&countrycodes=id&addressdetails=1';
    
    const response = UrlFetchApp.fetch(url, {
      headers: { 
        'Accept-Language': 'id',
        'User-Agent': 'YASSA-App/1.0'
      },
      muteHttpExceptions: true
    });
    
    const data = JSON.parse(response.getContentText());
    return { success: true, data: data };
  } catch(e) {
    return { success: false, message: e.message };
  }
}
